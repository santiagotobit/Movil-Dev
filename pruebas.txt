def test_add_and_remove_product(client):
    # Agregar producto
    response = client.post('/cart/add', json={
        "product_id": 1, "quantity": 2, "price": 50.0
    })
    assert response.status_code == 201

    # Calcular total
    response = client.get('/cart/total')
    data = response.get_json()
    assert data['subtotal'] == 100.0

    # Eliminar producto
    response = client.delete('/cart/remove/1')
    assert response.status_code == 200

# app.py
from flask import Flask, request, jsonify, session
from flask_sqlalchemy import SQLAlchemy

app = Flask(__name__)
app.config['SECRET_KEY'] = 'supersecret'
app.config['SQLALCHEMY_DATABASE_URI'] = 'postgresql://user:password@localhost/cartdb'
db = SQLAlchemy(app)

# Modelo de carrito
class CartItem(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    product_id = db.Column(db.Integer, nullable=False)
    quantity = db.Column(db.Integer, nullable=False)
    price = db.Column(db.Float, nullable=False)

@app.route('/cart/add', methods=['POST'])
def add_to_cart():
    data = request.json
    item = CartItem(product_id=data['product_id'],
                    quantity=data['quantity'],
                    price=data['price'])
    db.session.add(item)
    db.session.commit()
    return jsonify({"message": "Producto agregado"}), 201

@app.route('/cart/remove/<int:item_id>', methods=['DELETE'])
def remove_from_cart(item_id):
    item = CartItem.query.get(item_id)
    if not item:
        return jsonify({"error": "Item no encontrado"}), 404
    db.session.delete(item)
    db.session.commit()
    return jsonify({"message": "Producto eliminado"}), 200

@app.route('/cart/total', methods=['GET'])
def calculate_total():
    items = CartItem.query.all()
    subtotal = sum([i.price * i.quantity for i in items])
    tax = subtotal * 0.19  # IVA 19%
    shipping = 10.0        # tarifa fija
    total = subtotal + tax + shipping
    return jsonify({
        "subtotal": subtotal,
        "tax": tax,
        "shipping": shipping,
        "total": total
    })
